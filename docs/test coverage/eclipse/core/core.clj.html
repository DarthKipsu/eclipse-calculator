006  (defn- mapv-indexed [f v]
007    (vec (map-indexed f v)))
008  
009  (defn reform-ships
010    "Takes a vector containing ships involved in the battle, reforms them to be
011    usable for probability calculations and returns a new array containing the 
012    reformed ships."
013    [ships]
014    (mapv-indexed reform-single-ship ships))
015  
016  (defn win-probabilities
017    "Takes a vector containig map presentations of ships and returns a map contai-
018    ning win probabilities for both sides as well as vector containing probabilities
019    for each ship to be alive after the match."
020    [ships]
021    (let [ships-missiles (missiles-round ships)
022          ships-cannons (cannons-round ships-missiles 3)
023          defenders (targets-for "attacker" ships-cannons)
024          attackers (targets-for "defender" ships-cannons)
025          def-win-odds (win-odds-defender defenders attackers)
026          att-win-odds (win-odds-attacker def-win-odds defenders)]
027      {"defender" (format "%.1f" (* 100 def-win-odds))
028       "attacker" (format "%.1f" (* 100 att-win-odds))
029       "alive-odds" (alive-odds-formatted ships-cannons)}))
