003  (defn my-get-in
004    "takes a map and a vector containing a path inside that map as a parameter and
005    returns the value nested inside the sequence of keys in the given vector."
006    [a-map key-seq]
007    (loop [acc a-map i 0]
008      (if (= i (count key-seq))
009        acc
010        (recur ((key-seq i) acc) (inc i)))))
011  
012  (defn my-repeat
013    "takes a value of any type and a multiplier and returns a lazy sequence with
014    the value repeated by the multiplier."
015    ([n value] (my-repeat n value '()))
016    ([n value a-seq]
017     (if (zero? n)
018       a-seq
019       (recur (dec n) value (conj a-seq value)))))
020  
021  (defn to-vector
022    "takes a sequence and returns a vector containing all the elements in the seq."
023    [a-seq]
024    (if (vector? a-seq) a-seq (vec a-seq)))
025  
026  (defn my-first
027    "takes a sequence and returns the first element in it."
028    [a-seq]
029    (let [a-vec (to-vector a-seq)]
030      (a-vec 0)))
031  
032  (defn my-rest
033    "takes a sequence and returns a new vector with all but the first element of
034    the given sequence"
035    [a-seq]
036    (let [a-vec (to-vector a-seq)]
037      (loop [acc [] n 1]
038        (if (== n (count a-seq))
039          acc
040          (recur (conj acc (a-vec n)) (inc n))))))
041  
042  (defn apply-all 
043    "a helper function for my-map that takes a function and a sequence, goes through
044    the sequence and applies the function to each value and then returns the results
045    of the function in a sequence."
046    [func a-seq]
047    (persistent! (reduce (fn [acc n] (conj! acc (func n))) (transient []) a-seq)))
048    
049  (defn my-some
050    "takes a predicate function and a sequence and returns true when first of the
051    sequences values meets the predicate and nil if none will."
052    [func [seq-first & seq-rest]]
053    (when seq-first (or (func seq-first) (recur func seq-rest))))
054  
055  (defn my-map
056    "takes a function and one or more sequences and returns a vector containing
057    the results of applying the function to the first set of parameters, then the 
058    second and so on until reaching the end of one or more of the sequences."
059    ([func a-seq] (apply-all func a-seq))
060    ([func a-seq & more]
061     (if (or (empty? a-seq) (my-some empty? more))
062       []
063       (apply vector (apply func (my-first a-seq) (apply-all my-first more))
064              (apply my-map func (my-rest a-seq) (apply-all my-rest more))))))

066  (defn my-range
067    "returns a seq containing numbers starting from 0 until the number (exclusive)
068    given, or starting from the first parameter (inclusive) and ending before the
069    second one."
070    ([end] (my-range 0 end))
071    ([start end]
072     (loop [acc '() n (dec end)]
073       (if (== n (dec start)) acc
074         (recur (conj acc n) (dec n))))))
